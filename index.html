function doPost(e) {
  try {
    // Reads URL-encoded body: data=
    const raw  = e.parameter.data || e.postData.contents;
    const data = JSON.parse(raw);
    const ss   = SpreadsheetApp.getActiveSpreadsheet();
    let   sh   = ss.getSheetByName(SHEET_NAME);
    if (!sh) sh = ss.insertSheet(SHEET_NAME);
    const sites = data.sites   || {};
    const prof  = data.surveyor || {};

    // Write header row if sheet is empty
    if (sh.getLastRow() === 0) {
      const headers = ['Submitted At', 'Surveyor', 'Phone', '#'].concat(COLUMN_MAP.map(c => c[1]));
      sh.appendRow(headers);
      sh.getRange(1, 1, 1, headers.length).setFontWeight('bold').setBackground('#1A6B3A').setFontColor('#ffffff');
      sh.setFrozenRows(1);
    }

    let rowNum = sh.getLastRow();
    Object.entries(sites).forEach(([sid, row]) => {
      rowNum++;
      const clean = {};
      Object.entries(row).forEach(([k, v]) => {
        const bare = k.replace(/^site_\d+_/, '').replace(/^[^_]+_/, '');
        clean[bare] = v;
      });
      const vals = [
        data.exported_at,
        prof.name  || '',
        prof.phone || '',
        rowNum - 1,
      ].concat(COLUMN_MAP.map(([key]) => clean[key] ?? ''));
      sh.appendRow(vals);
    });

    return ContentService
      .createTextOutput(JSON.stringify({ ok: true, rows: Object.keys(sites).length }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet() {
  return ContentService.createTextOutput('Powerhive Spark Survey endpoint · active');
}
